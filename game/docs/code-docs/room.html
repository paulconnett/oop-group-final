196    def combat(self, player: Player, enemy: Enemy) -> None:
197        '''
198        combat engine
199
200        args:
201            player: Player - current player object
202            enemy: Enemy - current monster being fought
203        '''
204        while enemy.is_alive:
205            print(f"===== {player.name} - {player.hp} HP - {player.attack} ATK - "
206                  f"{player.defense} AMR =====")
207            print(f"===== {enemy.name} - {enemy.hp} HP - {enemy.attack} ATK - "
208                  f"{enemy.defense} AMR =====\n")
209
210            options = ("Attack", "Use potion")
211            for o in options:
212                print(f"{options.index(o) + 1} - {o}")
213
214            action = input(f"\nWhat will the brave {player.name} do? (Enter action number...) ")
215            os.system('clear')
216
217            if action == '1':
218                dmg = player.attack_entity(enemy)
219                print(f"{player.name} attacks for {dmg} damage!")
220
221            if action == '2':
222                self.use_potion(player)
223                continue
224
225            if enemy.is_alive:
226                dmg = enemy.attack_entity(player)
227                input(f"\nThe {enemy.name} attacks for {dmg} damage!    (Press enter...)")
228                os.system('clear')
229
230            if not player.is_alive:  # Need better game over but this will do for now
231                os.system('clear')
232                input("u ded")
233                sys.exit()
234
235        print(f"\nThe {enemy.name} has been defeated!")
236        player.add_money(enemy.reward)
237        input(f"{player.name} gained {enemy.reward}c!  (Press enter...)")
238
239    def use_potion(self, player: Player) -> None:
240        '''
241        player uses a potion from their inventory
242
243        args:
244            player: Player - current player object
245        '''
246        if not player.inventory:
247            input("No potions to use!   (Press enter...)")
248            return
249
250        potions: List[Potion] = []
251
252        print("===== Iventory =====")
253        for item in player.inventory:
254            if isinstance(item, Potion):
255                potions.append(item)
256
257        for item in potions:
258            print(f"{potions.index(item) + 1} - {item.name}")
259
260        choice = input(f"Which item will {player.name} use? (Enter item number...) ")
261
262        try:
263            choice_num = int(choice) - 1
264            player.use_potion(potions[choice_num])
265            os.system('clear')
266            print(f"{player.name} used a {potions[choice_num].name}")
267        except (IndexError, ValueError):
268            print("INVALID INPUT")
269            self.use_potion(player)
270        return
